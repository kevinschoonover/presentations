HTTP

Hypertext Transfer Protocol

HTTP Methods

Method Description
GET Read data
POST Insert data
PATCH or PUT Modify existing data or insert data if new
DELETE Remove data

Demo 6

HTTP Methods


from flask import Flask, request
app = Flask(__name__)


@app.route("/", methods=['GET', 'POST', 'PATCH', 'DELETE'])
def hello():
    if request.method == 'GET':
        return 'GET Request'
    elif request.method == 'POST':
        return 'POST Request'
    elif request.method == 'PATCH':
        return 'PATCH Request'
    elif request.method == 'DELETE':
        return 'DELETE Request'
    else:
        return 'Request'

Demo 8

Variable Rules


from flask import Flask, request
app = Flask(__name__)


@app.route("/")
def index():
    return "/users/<br />/posts/<br />/folders/"


@app.route("/users/<username>")
def user(username):
    return "User {}".format(username)


@app.route("/posts/<int:id>/")
def post(id):
    return "Post #{}".format(id)


@app.route("/folders/<path:subpath>/")
def folder(subpath):
    return "Folder is {}".format(subpath)
